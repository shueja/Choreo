20static constexpr frc::Pose2d Interpolate(const frc::Pose2d& startValue,
21 const frc::Pose2d& endValue,
22 double t) {
23 if (t < 0) {
24 return startValue;
25 } else if (t >= 1) {
26 return endValue;
27 } else {
28 frc::Twist2d twist{startValue.Log(endValue)};
29 frc::Twist2d scaledTwist = twist * t;
30 return startValue.Exp(scaledTwist);
31 }
32}

41template <typename T>
42concept EqualityComparable = requires(const T& a, const T& b) {
43 { a == b } -> std::convertible_to<bool>;
44 { a != b } -> std::convertible_to<bool>;
45};

50template <typename T>
