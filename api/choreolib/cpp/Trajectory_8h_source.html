216 std::optional<Trajectory<SampleType>> GetSplit(int splitIndex) const {
217 if (splitIndex < 0 || splitIndex >= splits.size()) {
218 return std::nullopt;
219 }
220
221 int start = splits[splitIndex];
222 int end = (splitIndex + 1 < splits.size()) ? splits[splitIndex + 1] + 1
223 : samples.size();
224
225 auto sublist =
226 std::vector<SampleType>(samples.begin() + start, samples.begin() + end);
227 units::second_t startTime = sublist.front().GetTimestamp();
228 units::second_t endTime = sublist.back().GetTimestamp();
229
230 auto offsetSamples =
231 sublist | std::views::transform([startTime](const SampleType& s) {
232 return s.OffsetBy(-startTime);
233 });
234
235 auto filteredEvents =
236 events | std::views::filter([startTime, endTime](const auto& e) {
237 return e.timestamp >= startTime && e.timestamp <= endTime;
238 }) |
239 std::views::transform(
240 [startTime](const auto& e) { return e.OffsetBy(-startTime); });
241
242 return Trajectory<SampleType>{
243 name + "[" + std::to_string(splitIndex) + "]",
244 std::vector<SampleType>(offsetSamples.begin(), offsetSamples.end()),
245 {},
246 std::vector<EventMarker>(filteredEvents.begin(), filteredEvents.end())};
247 }

294 std::optional<SampleType> SampleInternal(units::second_t timestamp) const {
295 if (samples.size() == 0) {
296 return {};
297 }
298 if (samples.size() == 1) {
299 return samples[0];
300 }
301 if (timestamp < samples[0].GetTimestamp()) {
302 return GetInitialSample();
303 }
304 if (timestamp >= GetTotalTime()) {
305 return GetFinalSample();
306 }
307
308 int low = 0;
309 int high = samples.size() - 1;
310
311 while (low != high) {
312 int mid = (low + high) / 2;
313 if (samples[mid].GetTimestamp() < timestamp) {
314 low = mid + 1;
315 } else {
316 high = mid;
317 }
318 }
319
320 if (low == 0) {
321 return samples[low];
322 }
323
324 SampleType behindState = samples[low - 1];
325 SampleType aheadState = samples[low];
326
327 if ((aheadState.GetTimestamp() - behindState.GetTimestamp()) < 1e-6_s) {
328 return aheadState;
329 }
330
331 return behindState.Interpolate(aheadState, timestamp);
332 }
