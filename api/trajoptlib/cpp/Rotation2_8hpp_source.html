136template <typename T, typename U>
137 requires std::same_as<T, sleipnir::Variable> ||
138 std::same_as<U, sleipnir::Variable>
139sleipnir::EqualityConstraints operator==(const Rotation2<T>& lhs,
140 const Rotation2<U>& rhs) {
141 std::vector<sleipnir::EqualityConstraints> constraints;
142
143 // Constrain angle equality on manifold.
144 //
145 // Let lhs = <cos(a), sin(a)>. NOLINT
146 // Let rhs = <cos(b), sin(b)>. NOLINT
147 //
148 // If the angles are equal, the angle between the unit vectors should be
149 // zero.
150 //
151 // lhs x rhs = ‖lhs‖ ‖rhs‖ sin(angleBetween) NOLINT
152 // = 1 * 1 * 0
153 // = 0
154 //
155 // NOTE: angleBetween = π rad would be another solution
156 constraints.emplace_back(lhs.Cos() * rhs.Sin() - lhs.Sin() * rhs.Cos() ==
157 0.0);
158
159 // Require that lhs and rhs are unit vectors if they contain autodiff
160 // variables, since their values can change.
161 if constexpr (std::same_as<T, sleipnir::Variable>) {
162 constraints.emplace_back(lhs.Cos() * lhs.Cos() + lhs.Sin() * lhs.Sin() ==
163 1.0);
164 }
165 if constexpr (std::same_as<U, sleipnir::Variable>) {
166 constraints.emplace_back(rhs.Cos() * rhs.Cos() + rhs.Sin() * rhs.Sin() ==
167 1.0);
168 }
169
170 return sleipnir::EqualityConstraints{constraints};
171}
