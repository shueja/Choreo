46 static ProjectFile GetProjectFile() {
47 if (LAZY_PROJECT_FILE.has_value()) {
48 return LAZY_PROJECT_FILE.value();
49 }
50
51 std::vector<std::filesystem::path> matchingFiles;
52
53 try {
54 for (const auto& entry :
55 std::filesystem::directory_iterator(CHOREO_DIR)) {
56 if (std::filesystem::is_regular_file(entry) &&
57 entry.path().extension() == TRAJECTORY_FILE_EXTENSION) {
58 matchingFiles.push_back(entry.path());
59 }
60 }
61
62 if (matchingFiles.size() == 0) {
63 FRC_ReportError(frc::warn::Warning,
64 "Could not find file in deploy directory!");
65 } else if (matchingFiles.size() > 1) {
66 FRC_ReportError(frc::warn::Warning,
67 "Found multiple project files in deploy directory!");
68 }
69
70 auto fileBuffer = wpi::MemoryBuffer::GetFile(matchingFiles[0].string());
71 if (!fileBuffer) {
72 FRC_ReportError(frc::warn::Warning,
73 "Could not open choreo project file");
74 }
75
76 wpi::json json = wpi::json::parse(fileBuffer.value()->GetCharBuffer());
77 uint32_t version = json["version"];
78 if (kChorSpecVersion != version) {
79 throw fmt::format(".chor project file: Wrong version {}. Expected {}",
80 version, kChorSpecVersion);
81 }
82 ProjectFile resultProjectFile;
83 from_json(json, resultProjectFile);
84 LAZY_PROJECT_FILE = resultProjectFile;
85 } catch (const std::filesystem::filesystem_error&) {
86 FRC_ReportError(frc::warn::Warning, "Error finding choreo directory!");
87 } catch (const wpi::json::exception&) {
88 FRC_ReportError(frc::warn::Warning, "Error parsing choreo project file!");
89 }
90 return LAZY_PROJECT_FILE.value();
91 }

103 template <TrajectorySample SampleType>
104 static std::optional<Trajectory<SampleType>> LoadTrajectory(
105 std::string_view trajectoryName) {
106 if (trajectoryName.ends_with(TRAJECTORY_FILE_EXTENSION)) {
107 trajectoryName = trajectoryName.substr(
108 0, trajectoryName.size() - TRAJECTORY_FILE_EXTENSION.size());
109 }
110
111 std::string trajectoryFileName = fmt::format(
112 "{}/{}{}", CHOREO_DIR, trajectoryName, TRAJECTORY_FILE_EXTENSION);
113
114 auto fileBuffer = wpi::MemoryBuffer::GetFile(trajectoryFileName);
115 if (!fileBuffer) {
116 FRC_ReportError(frc::warn::Warning, "Could not find trajectory file: {}",
117 trajectoryName);
118 return {};
119 }
120
121 try {
122 return LoadTrajectoryString<SampleType>(
123 std::string{fileBuffer.value()->GetCharBuffer().data(),
124 fileBuffer.value()->size()},
125 trajectoryName);
126 } catch (wpi::json::parse_error& ex) {
127 FRC_ReportError(frc::warn::Warning, "Could not parse trajectory file: {}",
128 trajectoryName);
129 FRC_ReportError(frc::warn::Warning, "{}", ex.what());
130 return {};
131 }
132 return {};
133 }

145 template <TrajectorySample SampleType>
146 static std::optional<Trajectory<SampleType>> LoadTrajectoryString(
147 std::string_view trajectoryJsonString, std::string_view trajectoryName) {
148 if constexpr (std::same_as<SampleType, SwerveSample>) {
149 HAL_Report(HALUsageReporting::kResourceType_ChoreoTrajectory, 1);
150 } else if constexpr (std::same_as<SampleType, DifferentialSample>) {
151 HAL_Report(HALUsageReporting::kResourceType_ChoreoTrajectory, 2);
152 }
153
154 wpi::json json = wpi::json::parse(trajectoryJsonString);
155 uint32_t version = json["version"];
156 if (version != kTrajSpecVersion) {
157 throw fmt::format("{}.traj: Wrong version {}. Expected {}",
158 trajectoryName, version, kTrajSpecVersion);
159 }
160 Trajectory<SampleType> trajectory;
161 from_json(json, trajectory);
162 return trajectory;
163 }

169 template <TrajectorySample SampleType>
170 class TrajectoryCache {
171 public:
185 static std::optional<Trajectory<SampleType>> LoadTrajectory(
186 std::string_view trajectoryName) {
187 if (cache.contains(std::string{trajectoryName})) {
188 return cache[std::string{trajectoryName}];
189 } else {
190 cache[std::string{trajectoryName}] =
191 Choreo::LoadTrajectory<SampleType>(trajectoryName);
192 return cache[std::string{trajectoryName}];
193 }
194 }
195
211 static std::optional<Trajectory<SampleType>> LoadTrajectory(
212 std::string_view trajectoryName, int splitIndex) {
213 std::string key = fmt::format("{}.:.{}", trajectoryName, splitIndex);
214
215 if (!cache.contains(key)) {
216 if (cache.contains(std::string{trajectoryName})) {
217 cache[key] =
218 cache[std::string{trajectoryName}].value().GetSplit(splitIndex);
219 } else {
220 auto possibleTrajectory = LoadTrajectory(trajectoryName);
221 cache[std::string{trajectoryName}] = possibleTrajectory;
222
223 if (possibleTrajectory.has_value()) {
224 cache[key] = possibleTrajectory.value().GetSplit(splitIndex);
225 }
226 }
227 }
228
229 return cache[key];
230 }
231
235 static void Clear() { cache.clear(); }
236
237 private:
238 static inline std::unordered_map<std::string,
239 std::optional<Trajectory<SampleType>>>
240 cache;
241 };
