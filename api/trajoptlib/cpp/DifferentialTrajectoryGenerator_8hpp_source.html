153class TRAJOPT_DLLEXPORT DifferentialTrajectory {
154 public:
156 std::vector<DifferentialTrajectorySample> samples;
157
158 DifferentialTrajectory() = default;
159
165 explicit DifferentialTrajectory(
166 std::vector<DifferentialTrajectorySample> samples)
167 : samples{std::move(samples)} {}
168
174 explicit DifferentialTrajectory(const DifferentialSolution& solution) {
175 double ts = 0.0;
176 for (size_t sample = 0; sample < solution.x.size(); ++sample) {
177 samples.emplace_back(
178 ts, solution.x[sample], solution.y[sample], solution.heading[sample],
179 solution.vl[sample], solution.vr[sample], solution.al[sample],
180 solution.ar[sample], solution.Fl[sample], solution.Fr[sample]);
181 ts += solution.dt[sample];
182 }
183 }
184};
