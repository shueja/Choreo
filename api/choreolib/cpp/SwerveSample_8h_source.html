109 template <int Year>
110 constexpr SwerveSample Flipped() const {
111 constexpr auto flipper = choreo::util::GetFlipperForYear<Year>();
112 if constexpr (flipper.isMirrored) {
113 return SwerveSample{timestamp,
114 flipper.FlipX(x),
115 y,
116 flipper.FlipHeading(heading),

179 constexpr SwerveSample Interpolate(const SwerveSample& endValue,
180 units::second_t t) const {
181 units::scalar_t scale = (t - timestamp) / (endValue.timestamp - timestamp);
182 frc::Pose2d interpolatedPose =
183 frc::Interpolate(GetPose(), endValue.GetPose(), scale.value());
184
185 std::array<units::newton_t, 4> interpolatedForcesX;
186 std::array<units::newton_t, 4> interpolatedForcesY;
187 for (int i = 0; i < 4; i++) {
188 interpolatedForcesX[i] =
189 wpi::Lerp(moduleForcesX[i], endValue.moduleForcesX[i], scale.value());
190 interpolatedForcesY[i] =
191 wpi::Lerp(moduleForcesY[i], endValue.moduleForcesY[i], scale.value());
192 }
193
194 return SwerveSample{wpi::Lerp(timestamp, endValue.timestamp, scale),
195 interpolatedPose.X(),
196 interpolatedPose.Y(),
197 interpolatedPose.Rotation().Radians(),
198 wpi::Lerp(vx, endValue.vx, scale),
199 wpi::Lerp(vy, endValue.vy, scale),
200 wpi::Lerp(omega, endValue.omega, scale),
201 wpi::Lerp(ax, endValue.ax, scale),
202 wpi::Lerp(ay, endValue.ay, scale),
203 wpi::Lerp(alpha, endValue.alpha, scale),
204 interpolatedForcesX,
205 interpolatedForcesY};
206 }

214 constexpr bool operator==(const SwerveSample& other) const {
215 constexpr double epsilon = 1e-6;
216
217 auto compare_units = [epsilon](const auto& a, const auto& b) {
218 using UnitType =
219 std::remove_const_t<std::remove_reference_t<decltype(a)>>;
220 return units::math::abs(a - b) < UnitType(epsilon);
221 };
222
223 auto compare_arrays = [&compare_units](const auto& arr1, const auto& arr2) {
224 return std::equal(arr1.begin(), arr1.end(), arr2.begin(), compare_units);
225 };
226
227 return compare_units(timestamp, other.timestamp) &&
228 compare_units(x, other.x) && compare_units(y, other.y) &&
229 compare_units(heading, other.heading) &&
230 compare_units(vx, other.vx) && compare_units(vy, other.vy) &&
231 compare_units(omega, other.omega) && compare_units(ax, other.ax) &&
232 compare_units(ay, other.ay) && compare_units(alpha, other.alpha) &&
233 compare_arrays(moduleForcesX, other.moduleForcesX) &&
234 compare_arrays(moduleForcesY, other.moduleForcesY);
235 }
