179 void SgmtConstraint(size_t fromIndex, size_t toIndex,
180 const Constraint& constraint) {
181 assert(fromIndex < toIndex);
182
183 NewWpts(toIndex);
184 path.waypoints.at(fromIndex).waypointConstraints.push_back(constraint);
185 for (size_t index = fromIndex + 1; index <= toIndex; ++index) {
186 path.waypoints.at(index).waypointConstraints.push_back(constraint);
187 path.waypoints.at(index).segmentConstraints.push_back(constraint);

242 void NewWpts(size_t finalIndex) {
243 if (finalIndex < path.waypoints.size()) {
244 return;
245 }
246
247 for (size_t i = path.waypoints.size(); i <= finalIndex; ++i) {
248 path.waypoints.emplace_back();
249 initialGuessPoints.emplace_back(std::vector<Pose2d>{{0.0, 0.0, {0.0}}});
250 if (i != 0) {
251 controlIntervalCounts.push_back(40);
252 }
253 }
254 }
