109 constexpr DifferentialSample Interpolate(const DifferentialSample& endValue,
110 units::second_t t) const {
111 units::scalar_t scale = (t - timestamp) / (endValue.timestamp - timestamp);
112 frc::Pose2d interpolatedPose =
113 frc::Interpolate(GetPose(), endValue.GetPose(), scale.value());
114
115 return DifferentialSample{
116 wpi::Lerp(timestamp, endValue.timestamp, scale),
117 interpolatedPose.X(),
118 interpolatedPose.Y(),
119 interpolatedPose.Rotation().Radians(),
120 wpi::Lerp(vl, endValue.vl, scale),
121 wpi::Lerp(vr, endValue.vr, scale),
122 wpi::Lerp(al, endValue.al, scale),
123 wpi::Lerp(ar, endValue.ar, scale),
124 wpi::Lerp(fl, endValue.fl, scale),
125 wpi::Lerp(fr, endValue.fr, scale),
126 };
127 }

145 fr, fl);

155 constexpr bool operator==(const DifferentialSample& other) const {
156 constexpr double epsilon = 1e-6;
157
158 auto compare_units = [epsilon](const auto& a, const auto& b) {
159 using UnitType =
160 std::remove_const_t<std::remove_reference_t<decltype(a)>>;
161 return units::math::abs(a - b) < UnitType(epsilon);
162 };
163
164 return compare_units(timestamp, other.timestamp) &&
165 compare_units(x, other.x) && compare_units(y, other.y) &&
166 compare_units(heading, other.heading) &&
167 compare_units(vl, other.vl) && compare_units(vr, other.vr) &&
168 compare_units(al, other.al) && compare_units(ar, other.ar) &&
169 compare_units(fl, other.fl) && compare_units(fr, other.fr);
170 }
