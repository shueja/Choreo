186 explicit SwerveTrajectory(std::vector<SwerveTrajectorySample> samples)
187 : samples{std::move(samples)} {}
188
194 explicit SwerveTrajectory(const SwerveSolution& solution) {
195 double ts = 0.0;
196 for (size_t sample = 0; sample < solution.x.size(); ++sample) {
197 samples.emplace_back(
198 ts, solution.x[sample], solution.y[sample],
199 std::atan2(solution.thetasin[sample], solution.thetacos[sample]),
200 solution.vx[sample], solution.vy[sample], solution.omega[sample],
201 solution.ax[sample], solution.ay[sample], solution.alpha[sample],
202 solution.moduleFx[sample], solution.moduleFy[sample]);
203 ts += solution.dt[sample];
204 }
205 }
206};
